RISC-V¶

EX-01 RISC-V 代码理解¶

.data # 伪指令（仅在编译阶段用，不对应机器语言），接下来可定义数据段，初始化数据，数据标签化
.word 2, 4, 6, 8# 伪指令，在内存中分配和初始化 32 位宽的数据，这里为2，4，6，8
n: .word 9# 定义了一个标签为 n 的 32 位整数常量,其值为 9。这个常量也会被存储在内存的数据段中。

.text # 伪指令，接下来定义代码段
main: # 这是程序的入口点,也就是程序开始执行的位置。
    add t0, x0, x0
    addi t1, x0, 1
    la t3, n # la : load address
    lw t3, 0(t3)
fib: # 这是一个标签,表示斐波那契数列的计算循环。
    beq t3, x0, finish
    add t2, t1, t0
    mv t0, t1
    mv t1, t2
    addi t3, t3, -1
    j fib
finish:
    addi a0, x0, 1
    addi a1, t0, 0
    ecall # print integer ecall
    addi a0, x0, 10
    ecall # terminate ecall

EX-03 Loop and Recursion¶

以factorial为例下展示loop和recursion

Loop

.globl factorial

.data
n: .word 8

.text
main:
    la t0, n
    lw a0, 0(t0)
    jal ra, factorial

    addi a1, a0, 0
    addi a0, x0, 1
    ecall # Print Result

    addi a1, x0, '\n'
    addi a0, x0, 11
    ecall # Print newline

    addi a0, x0, 10
    ecall # Exit

factorial:
    # YOUR CODE HERE
    add t0, a0, x0
    addi sp, sp, -4
    sw ra, 0(sp)
    addi a1, a0, -1
loop:
    beq a1, x0, exit
    jal ra, mult
    addi a1, a1, -1
    jal x0, loop
exit:
    lw ra, 0(sp)
    addi sp, sp, 4
    jr ra

mult:
    add t1, a1, x0
    add t2, a0, x0
    addi t3, x0, 1
newloop:
    beq t1, t3, newexit
    add a0, a0, t2
    addi t1, t1, -1
    jal x0, newloop
newexit:
    jr ra

Recursion

.globl factorial

.data
n: .word 8

.text
main:
    la t0, n
    lw a0, 0(t0)
    jal ra, factorial

    addi a1, a0, 0
    addi a0, x0, 1
    ecall # Print Result

    addi a1, x0, '\n'
    addi a0, x0, 11
    ecall # Print newline

    addi a0, x0, 10
    ecall # Exit

factorial:
    # YOUR CODE HERE
li t0, 1
    beq a0, t0, finish # if (n==1) return 1
    addi sp, sp, -8
    sw ra, 4(sp)       # push ra
    sw a0, 0(sp)       # push a0
    addi a0, a0, -1
    jal ra, factorial  # call factorial(n-1)
    lw t0, 0(sp)       # pop a0
    mul a0, a0, t0     
    lw ra, 4(sp)       # pop ra
    addi sp, sp, 8
finish:
    ret

EX-04 Calling Convension¶

tip

all functions that overwrite registers that are preserved by convention must have a prologue, which saves those register values to the stack at the start of the function, and an epilogue, which restores those values for the function’s caller.

ra在递归函数时才要存到内存

所有寄存器无本质差别

Register Usage¶

eg: 

helper_fn:
    # BEGIN PROLOGUE
    addi sp, sp, -4
    sw s0, 0(sp)
    # END PROLOGUE
    lw t1, 0(a0)
    addi s0, t1, 1
    sw s0, 0(a0)
    # BEGIN EPILOGUE
    lw s0, 0(sp)
    addi sp, sp, 4
    # END EPILOGUE
    ret
